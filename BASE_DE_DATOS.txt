-- TRIGGER DE CALCULO DE EDAD(AUN UTIL) --

USE GESTION_TALLER;

IF OBJECT_ID('TR_SGT_PERSONA_CALCULAR_EDAD', 'TR') IS NOT NULL
    DROP TRIGGER TR_SGT_PERSONA_CALCULAR_EDAD;
GO

CREATE TRIGGER TR_SGT_PERSONA_CALCULAR_EDAD
ON SGT_PERSONA_TALLER
AFTER INSERT, UPDATE
AS BEGIN
	SET NOCOUNT ON;
	--servira para contar la edad del usuario calculando segun su nacimiento.
	UPDATE P
	SET EDAD = DATEDIFF(YEAR, I.FEC_NACIMIENTO, GETDATE())
	FROM SGT_PERSONA_TALLER P
		INNER JOIN INSERTED I ON P.ID_PERSONA = I.ID_PERSONA
	WHERE P.FEC_NACIMIENTO IS NOT NULL;
END;
GO

CREATE PROCEDURE VER_ESTUDIANTE
    @ID_ESTUDIANTE NUMERIC(10)
AS 
BEGIN
    SELECT 
        E.ID_TALLER,
        T.NOMBRE_TALLER,
        E.ID_ESTUDIANTE,
        E.YEAR_PROCESO,
        E.IND_ESTADO_INTEGRANTE,
        E.FEC_INSCRIPCION,
        E.FEC_RETIRO,
        E.OBSERVACION,
        E.FEC_REINCORPORACION,
        E.AUD_USUARIO_INGRESO,
        E.AUD_FEC_INGRESO,
        E.AUD_USUARIO_MODIFICA,
        E.AUD_FEC_MODIFICA
    FROM SGT_INTEGRANTE_TALLER E
    INNER JOIN SGT_TALLER T 
        ON E.ID_TALLER = T.ID_TALLER
    WHERE E.ID_ESTUDIANTE = @ID_ESTUDIANTE
    ORDER BY E.FEC_INSCRIPCION DESC;
END;

CREATE PROCEDURE VER_TALLERES
    @ID_TALLER NUMERIC(10)
AS 
BEGIN
    SELECT 
        A.ID_TALLER AS 'ID del taller',
        A.ID_CATEGORIA AS 'ID de la categoria',
        A.NOMBRE_TALLER AS 'Nombre del taller',
        A.ID_DEPARTAMENTO AS 'ID del departamento',
        A.OBJETIVO_TALLER AS 'Objetivo del taller',
        A.FEC_INICIO AS 'Fecha de inicio',
        A.FEC_TERMINO AS 'Fecha de termino',
        A.NRO_MINUTOS AS 'Numero de minutos',
        A.NRO_CLASES_ANUAL AS 'Numero de clases anuales',
        A.HORAS_TOTALES AS 'Horas totales',
        A.ID_ESTADO_TALLER AS 'ID del estado del taller',
        A.FEC_ESTADO_TALLER AS 'Fecha del estado del taller',
        A.OBSERVACION AS 'Observacion',
        A.LUGAR AS 'Lugar',
        A.MINIMO_ESTUDIANTE AS 'Minimo de estudiantes',
        A.MAXIMO_ESTUDIANTE AS 'Maximo de estudiantes',
        COUNT(E.ID_ESTUDIANTE) AS 'Personas inscritas',
        A.REQUISITO AS 'Requisitos',
        A.EDAD_MINIMA AS 'Edad minima',
        A.EDAD_MAXIMA AS 'Edad maxima',
        A.MATERIAL AS 'Materiales'
    FROM SGT_TALLER A
    LEFT JOIN SGT_INTEGRANTE_TALLER E
        ON A.ID_TALLER = E.ID_TALLER
    WHERE A.ID_ESTADO_TALLER IN (1,2)
      AND A.ID_TALLER = @ID_TALLER
    GROUP BY
        A.ID_TALLER,
        A.ID_CATEGORIA,
        A.NOMBRE_TALLER,
        A.ID_DEPARTAMENTO,
        A.OBJETIVO_TALLER,
        A.FEC_INICIO,
        A.FEC_TERMINO,
        A.NRO_MINUTOS,
        A.NRO_CLASES_ANUAL,
        A.HORAS_TOTALES,
        A.ID_ESTADO_TALLER,
        A.FEC_ESTADO_TALLER,
        A.OBSERVACION,
        A.LUGAR,
        A.MINIMO_ESTUDIANTE,
        A.MAXIMO_ESTUDIANTE,
        A.REQUISITO,
        A.EDAD_MINIMA,
        A.EDAD_MAXIMA,
        A.MATERIAL;
END;
GO

    DROP PROCEDURE INSCRIBIR_TALLER_O_ESPERA;

CREATE PROCEDURE INSCRIBIR_TALLER_O_ESPERA
    @ID_TALLER NUMERIC(10),
    @ID_ESTUDIANTE NUMERIC(10),
    @YEAR_PROCESO INT
AS 
BEGIN
    SET NOCOUNT OFF;
    
    DECLARE @CUPOS_DISPONIBLES INT;
    DECLARE @CUPOS_MAXIMOS INT;
    DECLARE @INSCRITOS INT;
    DECLARE @NUEVO_ID_ESPERA NUMERIC;
    DECLARE @MENSAJE VARCHAR(500);
    DECLARE @EN_LISTA_ESPERA BIT;

    SELECT @CUPOS_MAXIMOS = MAXIMO_ESTUDIANTE 
    FROM SGT_TALLER 
    WHERE ID_TALLER = @ID_TALLER;

    IF @CUPOS_MAXIMOS IS NULL
    BEGIN
        SELECT 'El taller especificado no existe' AS mensaje, 0 AS en_lista_espera, @ID_TALLER AS id_taller, @ID_ESTUDIANTE AS id_estudiante;
        RETURN;
    END

    SELECT @INSCRITOS = COUNT(*) 
    FROM SGT_INTEGRANTE_TALLER 
    WHERE ID_TALLER = @ID_TALLER 
    AND IND_ESTADO_INTEGRANTE = 1;

    SET @CUPOS_DISPONIBLES = @CUPOS_MAXIMOS - @INSCRITOS;
    IF EXISTS (SELECT 1 FROM SGT_INTEGRANTE_TALLER 
               WHERE ID_ESTUDIANTE = @ID_ESTUDIANTE 
               AND ID_TALLER = @ID_TALLER
               AND IND_ESTADO_INTEGRANTE = 1)
    BEGIN
        SET @MENSAJE = 'El estudiante ya está inscrito en este taller';
        SET @EN_LISTA_ESPERA = 0;
    END
    ELSE IF @CUPOS_DISPONIBLES > 0
    BEGIN
        INSERT INTO SGT_INTEGRANTE_TALLER (ID_TALLER, ID_ESTUDIANTE, YEAR_PROCESO, IND_ESTADO_INTEGRANTE, FEC_INSCRIPCION, FEC_RETIRO, OBSERVACION, FEC_REINCORPORACION, AUD_USUARIO_INGRESO, AUD_FEC_INGRESO, AUD_USUARIO_MODIFICA, AUD_FEC_MODIFICA) 
        VALUES (@ID_TALLER, @ID_ESTUDIANTE, @YEAR_PROCESO,1, GETDATE(), '1900-01-01 00:00:00.000', '-', '1900-01-01 00:00:00.000','sistema', GETDATE(), 'sistema', GETDATE());
        SET @MENSAJE = 'Inscripción exitosa';
        SET @EN_LISTA_ESPERA = 0;
    END
    ELSE
    BEGIN
        IF OBJECT_ID('SGT_LISTA_ESPERA', 'U') IS NOT NULL
        BEGIN
            INSERT INTO SGT_LISTA_ESPERA (ID_TALLER, ID_ESTUDIANTE, YEAR_PROCESO,IND_ESTADO_INTEGRANTE, FECHA_ESPERA, ESTADO) 
            VALUES (@ID_TALLER, @ID_ESTUDIANTE, @YEAR_PROCESO,1, GETDATE(), 'En Espera');
            SET @NUEVO_ID_ESPERA = SCOPE_IDENTITY();
            SET @MENSAJE = 'No hay cupos disponibles. Estudiante agregado a lista de espera (ID: ' + CAST(@NUEVO_ID_ESPERA AS VARCHAR) + ')';
            SET @EN_LISTA_ESPERA = 1;
        END
        ELSE
        BEGIN
            SET @MENSAJE = 'No hay cupos disponibles en el taller';
            SET @EN_LISTA_ESPERA = 1;
        END
    END;
    SELECT @MENSAJE AS mensaje, @EN_LISTA_ESPERA AS en_lista_espera, @ID_TALLER AS id_taller, @ID_ESTUDIANTE AS id_estudiante;
END;
GO
